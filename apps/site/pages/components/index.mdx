export const RegionModalUsage = () => {
  const { modal, openModal } = useUI()
  const [input, setInput] = useState('')
  return (
    <>
      <Button variant="primary" onClick={() => openModal()}>
        Open Region Modal
      </Button>
      {modal && (
        <RegionModal
          fadeOutOnSubmit={true}
          inputValue={input}
          onInput={(e) => setInput(e.currentTarget.value)}
        />
      )}
    </>
  )
}

export const SlideOverPartialRight = () => {
  const { modal, openModal } = useUI()
  return (
    <>
      <Button variant="primary" onClick={openModal} style={{ width: '9rem' }}>
        Right
      </Button>
      {modal && (
        <SlideOverUsage size="partial" direction="rightSide" isOpen={modal} />
      )}
    </>
  )
}
