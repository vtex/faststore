import path from 'path'
import { useSSG } from 'nextra/ssg'
import { Callout, Tab, Tabs } from 'nextra-theme-docs'
import PropsSection from 'site/components/PropsSection'
import { TokenTable, TokenRow, TokenDivider } from 'site/components/Tokens'
import { OverviewSection } from 'site/components/OverviewSection'
import { SectionItem, SectionList } from 'site/components/SectionItem'
import ImageGalleryUsage from 'site/components/ImageGallery/ImageGalleryUsage'
import { ImageGallery, ImageGallerySelector } from '@faststore/ui'
import { getComponentPropsFrom } from 'site/components/utilities/propsSection'

export const getStaticProps = () => {
  const imageGalleryPath = path.resolve(__filename)
  const components = ['ImageGallery.tsx', 'ImageGallerySelector.tsx']
  const [imageGalleryProps, imageGallerySelectorProps] = getComponentPropsFrom(
    imageGalleryPath,
    components
  )
  return {
    props: {
      // We add an `ssg` field to the page props,
      // which will be provided to the Nextra `useSSG` hook.
      ssg: {
        imageGalleryProps,
        imageGallerySelectorProps,
      },
    },
  }
}

export const ImageGalleryPropsSection = ({ component }) => {
  // Get the data from SSG, and render it as a component.
  const { imageGalleryProps, imageGallerySelectorProps } = useSSG()
  return {
    ImageGallery: <PropsSection propsList={imageGalleryProps} />,
    ImageGallerySelector: (
      <PropsSection propsList={imageGallerySelectorProps} />
    ),
  }[component]
}

<header>

# Image Gallery

An `ImageGallery` is used when you want to display a series of photos in a gallery on a post or page.

</header>

According to the quantity of `Image` to be displayed, the `ImageGallerySelector` will be displayed.

---

## Import

Import the component from [@faststore/ui](/../components)

```tsx
import {
  ImageGallery,
  ImageGallerySelector,
  ImageGalleryViewer,
} from '@faststore/ui'
```

Import Styles

```tsx
import '@faststore/ui/src/components/organisms/ImageGallery/styles.scss'
```

## Customization

For further customization, you can use the following data attributes:

`data-fs-image-gallery`

`data-fs-image-gallery="with-selector | without-selector"`

`data-fs-image-gallery-selector`

`data-fs-image-gallery-selector-control`

`data-fs-image-gallery-selector-control-button`

`data-fs-image-gallery-selector-elements`

`data-fs-image-gallery-selector-thumbnail`
