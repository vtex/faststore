## Import

Import the component from <a href="https://www.faststore.dev/reference/ui/get-started-faststore-ui">@faststore/ui</a>

```tsx
import { SlideOver, SlideOverHeader } from '@faststore/ui'
```

Import Styles

```tsx
import '@faststore/ui/src/components/organisms/SlideOver/styles.scss'
```

---

export const SlideOverBase = () => {
  const { modal, openModal } = useUI()
  return (
    <>
      <Button variant="primary" onClick={() => openModal()}>
        Open Modal
      </Button>
      {modal && (
        <SlideOver isOpen={modal} size="partial" direction="rightSide">
          <SlideOverHeader onClose={openModal}>SlideOver</SlideOverHeader>
          <main
            style={{
              display: 'flex',
              alignItems: 'center',
              justifyContent: 'space-between',
              padding: '1rem',
              height: '100%',
            }}
          >
            <p
              style={{
                margin: 'auto',
              }}
            >
              A generic <code>SlideOver</code> component.
            </p>
          </main>
        </SlideOver>
      )}
    </>
  )
}

export const SlideOverTemplate = () => (
  <UIProvider>
    <SlideOverBase />
  </UIProvider>
)
