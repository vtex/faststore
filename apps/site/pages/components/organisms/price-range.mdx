## Usage

export function Component() {
  const formatter = useCallback(
    (price) =>
      new Intl.NumberFormat('en-GB', {
        style: 'currency',
        currency: 'EUR',
      }).format(price),
    []
  )
  return (
    <PriceRange
      max={{ absolute: 500, selected: 250 }}
      min={{ absolute: 0, selected: 100 }}
      formatter={formatter}
      onEnd={(value) =>
        window.alert(
          `Min: ${formatter(value.min)}, Max: ${formatter(value.max)}`
        )
      }
    />
  )
}
