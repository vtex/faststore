export const getStaticProps = () => {
  const slideOverPath = path.resolve(__filename)
  const components = ['SlideOver.tsx', 'SlideOverHeader.tsx']
  const [slideOverProps, slideOverHeaderProps] = getComponentPropsFrom(
    slideOverPath,
    components
  )
  return {
    props: {
      // We add an `ssg` field to the page props,
      // which will be provided to the Nextra `useSSG` hook.
      ssg: {
        slideOverProps,
        slideOverHeaderProps,
      },
    },
  }
}

export const SlideOverPropsSection = ({ component }) => {
  // Get the data from SSG, and render it as a component.
  const { slideOverProps, slideOverHeaderProps } = useSSG()
  return {
    SlideOver: <PropsSection propsList={slideOverProps} />,
    SlideOverHeader: <PropsSection propsList={slideOverHeaderProps} />,
  }[component]
}
