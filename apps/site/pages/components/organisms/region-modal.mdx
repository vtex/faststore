import { Tab, Tabs } from 'nextra-theme-docs'
import PropsSection from 'site/components/PropsSection'
import { TokenTable, TokenRow, TokenDivider } from 'site/components/Tokens'
import { OverviewSection } from 'site/components/OverviewSection'
import { RegionModal, Button, UIProvider, useUI } from '@faststore/ui'
import { useState } from 'react'

import path from 'path'
import { useSSG } from 'nextra/ssg'
import { getComponentPropsFrom } from 'site/components/utilities/propsSection'

export const getStaticProps = () => {
  const regionModalPath = path.resolve(__filename)
  const components = ['RegionModal.tsx']
  const [regionModalProps] = getComponentPropsFrom(regionModalPath, components)
  return {
    props: {
      // We add an `ssg` field to the page props,
      // which will be provided to the Nextra `useSSG` hook.
      ssg: { regionModalProps },
    },
  }
}

export const RegionModalPropsSection = ({ component }) => {
  // Get the data from SSG, and render it as a component.
  const { regionModalProps } = useSSG()
  return {
    RegionModal: <PropsSection propsList={regionModalProps} />,
  }[component]
}

<header>

# Region Modal

It's a [Modal](/components/molecules/modal) with some region configurations. The component is part of [Regionalization](/components/feature-regionalization) feature.

</header>

export const RegionModalUsage = () => {
  const { modal, openModal } = useUI()
  const [input, setInput] = useState('')
  return (
    <>
      <Button variant="primary" onClick={() => openModal()}>
        Open Region Modal
      </Button>
      {modal && (
        <RegionModal
          fadeOutOnSubmit={true}
          inputValue={input}
          onInput={(e) => setInput(e.currentTarget.value)}
        />
      )}
    </>
  )
}
