import { Tab, Tabs } from 'nextra-theme-docs'
import PropsSection from 'site/components/PropsSection'
import { TokenTable, TokenRow } from 'site/components/Tokens'
import { OverviewSection } from 'site/components/OverviewSection'

import NewsletterUsage from 'site/components/Newsletter/NewsletterUsage'

import path from 'path'
import { useSSG } from 'nextra/ssg'
import { getComponentPropsFrom } from 'site/components/utilities/propsSection'

export const getStaticProps = () => {
  const newsletterPath = path.resolve(__filename)
  const components = [
    'Newsletter.tsx',
    'NewsletterAddendum.tsx',
    'NewsletterContent.tsx',
    'NewsletterForm.tsx',
    'NewsletterHeader.tsx',
  ]
  const [
    newsletterProps,
    newsletterAddendumProps,
    newsletterContentProps,
    newsletterFormProps,
    newsletterHeaderProps,
  ] = getComponentPropsFrom(newsletterPath, components)
  return {
    props: {
      // We add an `ssg` field to the page props,
      // which will be provided to the Nextra `useSSG` hook.
      ssg: {
        newsletterProps,
        newsletterAddendumProps,
        newsletterContentProps,
        newsletterFormProps,
        newsletterHeaderProps,
      },
    },
  }
}

export const NewsletterPropsSection = ({ component }) => {
  // Get the data from SSG, and render it as a component.
  const {
    newsletterProps,
    newsletterAddendumProps,
    newsletterContentProps,
    newsletterFormProps,
    newsletterHeaderProps,
  } = useSSG()
  return {
    Newsletter: <PropsSection propsList={newsletterProps} />,
    NewsletterAddendum: <PropsSection propsList={newsletterAddendumProps} />,
    NewsletterContent: <PropsSection propsList={newsletterContentProps} />,
    NewsletterForm: <PropsSection propsList={newsletterFormProps} />,
    NewsletterHeader: <PropsSection propsList={newsletterHeaderProps} />,
  }[component]
}

<header>

# Newsletter

The Newsletter component is a section with at least one input text with an action which its value could be submitted.

</header>

## Import

Import the components from [@faststore/ui](/../components)

```tsx
import {
  Newsletter,
  NewsletterAddendum,
  NewsletterContent,
  NewsletterForm,
  NewsletterHeader,
} from '@faststore/ui'
```

Import Styles

```tsx
import '@faststore/ui/src/components/organisms/Newsletter/styles.scss'
```

---

## Usage

Uses `NewsletterForm`, `NewsletterHeader` and `NewsletterContent` as compound components.

<Tabs items={['Example', 'Code']} defaultIndex="0">
  <Tab>
    <OverviewSection>
      <NewsletterUsage />
    </OverviewSection>
  </Tab>
  <Tab>
    ```tsx
    <Newsletter card colorVariant="main">
      <NewsletterForm onSubmit={() => null}>
        <NewsletterHeader
          icon={<Icon name="Envelope" />}
          title="Get News and Special Offers!"
          description="Receive our news and promotions in advance. Enjoy and get 10% off your first purchase. For more information click here."
        />
        <NewsletterContent>
          <InputField id="newsletter-name" required label="Name" />
          <InputField
            id="newsletter-email"
            type="email"
            required
            label="Email"
          />
          <NewsletterAddendum addendum="By subscribing to our newsletter you agree to to our Privacy Policy." />
          <Button
            variant="secondary"
            type="submit"
            inverse={colorVariant === 'main'}
          >
            Subscribe
          </Button>
        </NewsletterContent>
      </NewsletterForm>
    </Newsletter>
    ```
  </Tab>
</Tabs>
