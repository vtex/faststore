```tsx
import {
  SearchProductCard,
  SearchProductCardImage,
  SearchProductCardContent,
} from '@faststore/ui'
```

Import Styles

```tsx
import '@faststore/ui/src/components/molecules/SearchProductCard/styles.scss'
```
