import { Tab, Tabs } from 'nextra-theme-docs'
import PropsSection from 'site/components/PropsSection'
import { TokenTable, TokenRow, TokenDivider } from 'site/components/Tokens'
import { OverviewSection } from 'site/components/OverviewSection'
import { SearchAutoComplete, SearchAutoCompleteTerm, Icon } from '@faststore/ui'

import path from 'path'
import { useSSG } from 'nextra/ssg'
import { getComponentPropsFrom } from 'site/components/utilities/propsSection'

export const getStaticProps = () => {
  const searchAutoCompletePath = path.resolve(__filename)
  const components = ['SearchAutoComplete.tsx', 'SearchAutoCompleteTerm.tsx']
  const [searchAutoCompleteProps, searchAutoCompleteTermProps] =
    getComponentPropsFrom(searchAutoCompletePath, components)
  return {
    props: {
      // We add an `ssg` field to the page props,
      // which will be provided to the Nextra `useSSG` hook.
      ssg: {
        searchAutoCompleteProps,
        searchAutoCompleteTermProps,
      },
    },
  }
}

export const SearchAutoCompletePropsSection = ({ component }) => {
  // Get the data from SSG, and render it as a component.
  const { searchAutoCompleteProps, searchAutoCompleteTermProps } = useSSG()
  return {
    SearchAutoComplete: <PropsSection propsList={searchAutoCompleteProps} />,
    SearchAutoCompleteTerm: (
      <PropsSection propsList={searchAutoCompleteTermProps} />
    ),
  }[component]
}

<header>

# Search Auto Complete

List of auto completing suggestions based on searched term. This component is part of the [Search](/components/feature-search) feature.

</header>

## Import

Import the component from [@faststore/ui](/../components)

```tsx
import { SearchAutoComplete, SearchAutoCompleteTerm } from '@faststore/ui'
```

Import Styles

```tsx
import '@faststore/ui/src/components/molecules/SearchAutoComplete/styles.scss'
```

---

## Usage

export const searchContent = 'Appl'

export const suggestions = [{ value: 'apple airpods' }, { value: 'apple tv' }]

<Tabs items={['Example', 'Code']} defaultIndex="0">
  <Tab>
    <OverviewSection direction="column">
      Searching for: <code>{searchContent}</code>
      <SearchAutoComplete>
        {suggestions?.map((suggestion) => (
          <SearchAutoCompleteTerm
            key={suggestion.value}
            term={searchContent}
            suggestion={suggestion.value}
            linkProps={{ href: '#' }}
          />
        ))}
      </SearchAutoComplete>
    </OverviewSection>
  </Tab>
  <Tab>

    ```tsx
    const searchContent = 'Appl'

    const suggestions = [{ value: 'apple airpods' }, { value: 'apple tv' }]

    <SearchAutoComplete>
      {suggestions?.map((suggestion) => (
        <SearchAutoCompleteTerm
          key={suggestion.value}
          term={searchContent}
          suggestion={suggestion.value}
          linkProps={{ href: '#' }}
        />
      ))}
    </SearchAutoComplete>
    ```

  </Tab>
</Tabs>
