  const components = ['Gift.tsx', 'GiftImage.tsx', 'GiftContent.tsx']

      ssg: {
        giftProps,
        giftImageProps,
        giftContentProps,
      },
    },
  }
}

export const GiftPropsSection = ({ component }) => {
  // Get the data from SSG, and render it as a component.
  const { giftProps, giftImageProps, giftContentProps } = useSSG()
  return {
    Gift: <PropsSection propsList={giftProps} />,
    GiftImage: <PropsSection propsList={giftImageProps} />,
    GiftContent: <PropsSection propsList={giftContentProps} />,
  }[component]
}

<header>

# Gift

It displays an additional item to purchase as a gift.

</header>

The Gift component is a compound of the following:

- `Gift`: renders a `<div>` as wrapper with an `Icon`, `GiftImage`, and `GiftContent`.
- `GiftImage`: renders a `<div>` as wrapper with an `Image`.
- `GiftContent`: renders a `<div>` as wrapper with the gift information as children.

`data-fs-gift-image`

`data-fs-gift-content`
