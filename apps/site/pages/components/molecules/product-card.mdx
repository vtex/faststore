export const getStaticProps = () => {
  const productCardPath = path.resolve(__filename)
  const components = [
    'ProductCard.tsx',
    'ProductCardImage.tsx',
    'ProductCardContent.tsx',
  ]
  const [productCardProps, productCardImageProps, productCardContentProps] =
    getComponentPropsFrom(productCardPath, components)
  return {
    props: {
      // We add an `ssg` field to the page props,
      // which will be provided to the Nextra `useSSG` hook.
      ssg: {
        productCardProps,
        productCardImageProps,
        productCardContentProps,
      },
    },
  }
}

export const ProductCardPropsSection = ({ component }) => {
  // Get the data from SSG, and render it as a component.
  const { productCardProps, productCardImageProps, productCardContentProps } =
    useSSG()
  return {
    ProductCard: <PropsSection propsList={productCardProps} />,
    ProductCardImage: <PropsSection propsList={productCardImageProps} />,
    ProductCardContent: <PropsSection propsList={productCardContentProps} />,
  }[component]
}

<header>

# Product Card

Displays summarized information about a product. They usually present a call-to-action button, as well as the product's name, price, and image.

</header>

The `ProductCard` is a compound of the following:

- `ProductCardImage`: wraps the product's image.
- `ProductCardContent`: wraps the content's details.

## Use Cases

Use the `ProductCard` to:

- Present a product collection on the home page.
- Present product details on PLPs.
- Build a product slider that showcases multiple products at once, making it easier for visitors to browse your online store.
