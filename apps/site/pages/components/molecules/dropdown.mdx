export const getStaticProps = () => {
  const dropdownPath = path.resolve(__filename)
  const components = ['Dropdown.tsx', 'DropdownItem.tsx', 'DropdownMenu.tsx', 'DropdownButton.tsx']
  const [dropdownProps, dropdownItemProps, dropdownMenuProps, dropdownButtonProps ] = getComponentPropsFrom(
    dropdownPath,
    components
  )
  return {
    props: {
      // We add an `ssg` field to the page props,
      // which will be provided to the Nextra `useSSG` hook.
      ssg: {
        dropdownProps,
        dropdownItemProps,
        dropdownMenuProps,
        dropdownButtonProps,
      },
    },
  }
}

export const DropdownPropsSection = ({ component }) => {
  // Get the data from SSG, and render it as a component.
  const { dropdownProps, dropdownItemProps, dropdownMenuProps, dropdownButtonProps } = useSSG()

    DropdownItem: <PropsSection propsList={dropdownItemProps} />,
    DropdownMenu: <PropsSection propsList={dropdownMenuProps} />,
    DropdownButton: <PropsSection propsList={dropdownButtonProps} />,
  }[component]
}

## Customization

For further customization, you can use the following data attributes:

#### Dropdown Item

`data-fs-dropdown-item`

#### Dropdown Menu

`data-fs-dropdown-menu`

`data-fs-dropdown-overlay`

`data-fs-dropdown-menu`

`data-fs-dropdown-menu-size="small | regular"`

#### Dropdown Button

`data-fs-dropdown-button`

This component inherits [Button](/components/atoms/button) CSS selectors and styles.

### Keyboard Interactions

<br />

<Table>
  <TableHead>
    <TableRow>
      <TableCell scope="col" variant="header" align="left">
        Key
      </TableCell>
      <TableCell scope="col" variant="header" align="left">
        Description
      </TableCell>
    </TableRow>
  </TableHead>
  <TableBody>
    <TableRow>
      <TableCell align="left">
        <code>Enter</code>
      </TableCell>
      <TableCell align="left">
        When focusing on <code>DropdownButton</code>, <br />
        the Dropdown menu opens and focuses the first <code>DropdownItem</code>
      </TableCell>
    </TableRow>
    <TableRow>
      <TableCell align="left">
        <code>ArrowDown</code>
      </TableCell>
      <TableCell align="left">
        When focusing on an item, moves focus to the next{' '}
        <code>DropdownItem</code>
      </TableCell>
    </TableRow>
    <TableRow>
      <TableCell align="left">
        <code>ArrowUp</code>
      </TableCell>
      <TableCell align="left">
        When focusing on an item, moves focus to the previous{' '}
        <code>DropdownItem</code>
      </TableCell>
    </TableRow>
    <TableRow>
      <TableCell align="left">
        <code>Escape</code>
      </TableCell>
      <TableCell align="left">
        Closes <code>DropdownMenu</code> and moves focus to{' '}
        <code>DropdownButton</code>
      </TableCell>
    </TableRow>
    <TableRow>
      <TableCell align="left">
        <code>Home</code>
      </TableCell>
      <TableCell align="left">
        When <code>DropdownMenu</code> is open, move focus to the first{' '}
        <code>DropdownItem</code>
      </TableCell>
    </TableRow>
    <TableRow>
      <TableCell align="left">
        <code>End</code>
      </TableCell>
      <TableCell align="left">
        When <code>DropdownMenu</code> is open, move focus to the last{' '}
        <code>DropdownItem</code>
      </TableCell>
    </TableRow>
  </TableBody>
</Table>
