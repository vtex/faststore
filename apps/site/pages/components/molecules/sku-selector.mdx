import path from 'path'
import { useSSG } from 'nextra/ssg'
import { Tab, Tabs } from 'nextra-theme-docs'
import PropsSection from 'site/components/PropsSection'
import { TokenTable, TokenRow, TokenDivider } from 'site/components/Tokens'
import { OverviewSection } from 'site/components/OverviewSection'
import {
  SkuSelectorImage,
  SkuSelectorLabel,
  SkuSelectorColor,
} from 'site/components/SkuSelector'
import { getComponentPropsFrom } from 'site/components/utilities/propsSection'

import { product } from '../../../mocks/product'

export const getStaticProps = () => {
  const skuSelectorPath = path.resolve(__filename)
  const components = ['SkuSelector.tsx']
  const [skuSelectorProps] = getComponentPropsFrom(skuSelectorPath, components)
  return {
    props: {
      // We add an `ssg` field to the page props,
      // which will be provided to the Nextra `useSSG` hook.
      ssg: {
        skuSelectorProps,
      },
    },
  }
}

export const SkuSelectorPropsSection = ({ component }) => {
  // Get the data from SSG, and render it as a component.
  const { skuSelectorProps } = useSSG()
  return {
    SkuSelector: <PropsSection propsList={skuSelectorProps} />,
  }[component]
}

<header>

# Sku Selector

The `SkuSelector` component is mainly used in Product Details Pages (PDPs) to display all the available variants for a given product.

</header>

## Overview

This is a product related component, we recommend testing it using a product mock.

<Tabs items={['Example', 'Code']} defaultIndex="0">
  <Tab>
    <OverviewSection>
      <SkuSelectorImage />
      <SkuSelectorLabel />
      <SkuSelectorColor />
    </OverviewSection>
  </Tab>
  <Tab>
    ```tsx
    <SkuSelector
      skuPropertyName="Image"
      availableVariations={
        product.isVariantOf.skuVariants.availableVariations
      }
      activeVariations={product.isVariantOf.skuVariants.activeVariations}
      slugsMap={product.isVariantOf.skuVariants.slugsMap}
    />
    ```


## Customization

`data-fs-sku-selector`

`data-fs-sku-selector-variant= "image" | "label" | "color"`

`data-fs-sku-selector-title`

`data-fs-sku-selector-list`

`data-fs-sku-selector-disabled="true"`

`data-fs-sku-selector-checked="true"`

`data-fs-sku-selector-option`

`data-fs-sku-selector-option-image`

`data-fs-sku-selector-option-color`
