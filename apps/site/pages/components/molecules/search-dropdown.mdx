import { Tab, Tabs } from 'nextra-theme-docs'
import PropsSection from 'site/components/PropsSection'
import { TokenTable, TokenRow, TokenDivider } from 'site/components/Tokens'
import { OverviewSection } from 'site/components/OverviewSection'

import SearchDropdownUsage from 'site/components/Search/SearchDropdownUsage'
import { product } from 'site/mocks/product'

import path from 'path'
import { useSSG } from 'nextra/ssg'
import { getComponentPropsFrom } from 'site/components/utilities/propsSection'

export const getStaticProps = () => {
  const searchDropdownPath = path.resolve(__filename)
  const components = ['SearchDropdown.tsx']
  const [searchDropdownProps] = getComponentPropsFrom(
    searchDropdownPath,
    components
  )
  return {
    props: {
      // We add an `ssg` field to the page props,
      // which will be provided to the Nextra `useSSG` hook.
      ssg: {
        searchDropdownProps,
      },
    },
  }
}

export const SearchDropdownPropsSection = ({ component }) => {
  // Get the data from SSG, and render it as a component.
  const { searchDropdownProps } = useSSG()
  return {
    SearchDropdown: <PropsSection propsList={searchDropdownProps} />,
  }[component]
}

# Search Dropdown

Combine `SearchHistory`, `SearchTop`, `SearchAutoComplete` and `SearchProducts` to show all options in `SearchInput`. This component is part of the [Search](/components/feature-search) feature.

## Import

Import the component from [@faststore/ui](/../components)

```tsx
import { SearchDropdown } from '@faststore/ui'
```

### Import Styles into your FastStore project

To apply the styles of this component in your FastStore project, import the following into your stylesheet:

```scss copy
@import '@faststore/ui/src/components/molecules/SearchDropdown/styles.scss';
```

Follow the instructions in the [Importing FastStore UI component styles](/docs/customizing/importing-styles-to-faststore-ui-components) tutorial.
