import { Tab, Tabs } from 'nextra-theme-docs'
import PropsSection from 'site/components/PropsSection'
import { TokenTable, TokenRow, TokenDivider } from 'site/components/Tokens'
import { OverviewSection } from 'site/components/OverviewSection'
import { SearchHistory, SearchHistoryTerm, Icon } from '@faststore/ui'

import path from 'path'
import { useSSG } from 'nextra/ssg'
import { getComponentPropsFrom } from 'site/components/utilities/propsSection'

export const getStaticProps = () => {
  const searchHistoryPath = path.resolve(__filename)
  const components = ['SearchHistory.tsx', 'SearchHistoryTerm.tsx']
  const [searchHistoryProps, searchHistoryTermProps] = getComponentPropsFrom(
    searchHistoryPath,
    components
  )
  return {
    props: {
      // We add an `ssg` field to the page props,
      // which will be provided to the Nextra `useSSG` hook.
      ssg: {
        searchHistoryProps,
        searchHistoryTermProps,
      },
    },
  }
}

export const SearchHistoryPropsSection = ({ component }) => {
  // Get the data from SSG, and render it as a component.
  const { searchHistoryProps, searchHistoryTermProps } = useSSG()
  return {
    SearchHistory: <PropsSection propsList={searchHistoryProps} />,
    SearchHistoryTerm: <PropsSection propsList={searchHistoryTermProps} />,
  }[component]
}

# Search History

## Import

Import the component from [@faststore/ui](/../components)

```tsx
import { SearchHistory, SearchHistoryTerm } from '@faststore/ui'
```

```tsx

```
