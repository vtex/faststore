import { Tab, Tabs } from 'nextra-theme-docs'
import PropsSection from 'site/components/PropsSection'
import { TokenTable, TokenRow, TokenDivider } from 'site/components/Tokens'
import { OverviewSection } from 'site/components/OverviewSection'
import {
  SearchProducts,
  SearchProductItem,
  SearchProductItemImage,
  SearchProductItemContent,
} from '@faststore/ui'

import { useFormattedPrice } from 'site/components/utilities/usePriceFormatter'
import { product } from 'site/mocks/product'

import path from 'path'
import { useSSG } from 'nextra/ssg'
import { getComponentPropsFrom } from 'site/components/utilities/propsSection'

export const getStaticProps = () => {
  const searchProductsPath = path.resolve(__filename)
  const components = [
    'SearchProducts.tsx',
    'SearchProductItem.tsx',
    'SearchProductItemImage.tsx',
    'SearchProductItemContent.tsx',
  ]
  const [
    searchProductsProps,
    searchProductItemProps,
    searchProductItemImageProps,
    searchProductItemContentProps,
  ] = getComponentPropsFrom(searchProductsPath, components)
  return {
    props: {
      // We add an `ssg` field to the page props,
      // which will be provided to the Nextra `useSSG` hook.
      ssg: {
        searchProductsProps,
        searchProductItemProps,
        searchProductItemImageProps,
        searchProductItemContentProps,
      },
    },
  }
}

export const SearchProductsPropsSection = ({ component }) => {
  // Get the data from SSG, and render it as a component.
  const {
    searchProductsProps,
    searchProductItemProps,
    searchProductItemImageProps,
    searchProductItemContentProps,
  } = useSSG()
  return {
    SearchProducts: <PropsSection propsList={searchProductsProps} />,
    SearchProductItem: <PropsSection propsList={searchProductItemProps} />,
    SearchProductItemImage: (
      <PropsSection propsList={searchProductItemImageProps} />
    ),
    SearchProductItemContent: (
      <PropsSection propsList={searchProductItemContentProps} />
    ),
  }[component]
}

  SearchProducts,

  SearchProductItemImage,
  SearchProductItemContent,

```tsx

```
