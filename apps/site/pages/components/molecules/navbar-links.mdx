import path from 'path'
import { useSSG } from 'nextra/ssg'
import { Callout, Tab, Tabs } from 'nextra-theme-docs'
import PropsSection from 'site/components/PropsSection'
import { TokenTable, TokenRow, TokenDivider } from 'site/components/Tokens'
import { OverviewSection } from 'site/components/OverviewSection'
import { SectionItem, SectionList } from 'site/components/SectionItem'
import { Navbar, UIProvider } from '@faststore/ui'
import { getComponentPropsFrom } from 'site/components/utilities/propsSection'
import NavbarLinksUsage from 'site/components/Navigation/NavbarLinksUsage'
import NavbarUsage from 'site/components/Navigation/NavbarUsage'
import NavbarSliderUsage from 'site/components/Navigation/NavbarSliderUsage'

export const getStaticProps = () => {
  const navbarLinksPath = path.resolve(__filename)
  const components = ['NavbarLinks.tsx', 'NavbarLinksList.tsx']
  const [navbarLinksProps, navbarLinksListProps] = getComponentPropsFrom(
    navbarLinksPath,
    components
  )
  return {
    props: {
      // We add an `ssg` field to the page props,
      // which will be provided to the Nextra `useSSG` hook.
      ssg: {
        navbarLinksProps,
        navbarLinksListProps,
      },
    },
  }
}

export const NavbarLinksPropsSection = ({ component }) => {
  // Get the data from SSG, and render it as a component.
  const { navbarLinksProps, navbarLinksListProps } = useSSG()
  return {
    NavbarLinks: <PropsSection propsList={navbarLinksProps} />,
    NavbarLinksList: <PropsSection propsList={navbarLinksListProps} />,
  }[component]
}

<header>

# Navbar Links

A wrapper used to display a set of navigation links that self adapts on mobile or desktop screens.

This component is part of the [Navigation](/components/feature-navigation) feature and must be used in the [Navbar](/components/organisms/navbar) context.

</header>

## Import

Import the component from [@faststore/ui](/../components)

```tsx
import {
  NavbarLinks,
  NavbarLinksList,
  NavbarLinksListItem,
} from '@faststore/ui'
```

Import Styles

```tsx
import '@faststore/ui/src/components/molecules/NavbarLinks/styles.scss'
```

---

## Usage

<Callout type="warning" emoji="⚠️">
  The mobile version can be achieved with
  [NavbarSlider](/components/organisms/navbar-slider) as wrapper.
</Callout>

<Tabs items={['Example', 'Code']} defaultIndex="0">
  <Tab>
    <OverviewSection containerStyle={{ display: 'block' }}>
      <Navbar>
        <NavbarLinksUsage />
      </Navbar>
    </OverviewSection>
  </Tab>
  <Tab>
    ```tsx
    const links = ['Office', 'Home Appliances', 'Computer and Software', 'Technology']

    <Navbar>
      <NavbarLinks>
        <div>
          <Button variant="tertiary" size="small">
            Set location
          </Button>
          <NavbarLinksList>
            {links.map((link) => (
              <NavbarLinksListItem key={link}>
                <Link variant="display" href="#">
                  {link}
                </Link>
              </NavbarLinksListItem>
            ))}
          </NavbarLinksList>
        </div>
      </NavbarLinks>
    </Navbar>
    ```

  </Tab>
</Tabs>
