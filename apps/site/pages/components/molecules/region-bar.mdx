import { Tab, Tabs } from 'nextra-theme-docs'
import PropsSection from 'site/components/PropsSection'
import { TokenTable, TokenRow, TokenDivider } from 'site/components/Tokens'
import { OverviewSection } from 'site/components/OverviewSection'
import {
  RegionBar,
  RegionModal,
  Button,
  UIProvider,
  useUI,
  Icon,
} from '@faststore/ui'

import path from 'path'
import { useSSG } from 'nextra/ssg'
import { getComponentPropsFrom } from 'site/components/utilities/propsSection'

export const getStaticProps = () => {
  const regionBarPath = path.resolve(__filename)
  const components = ['RegionBar.tsx']
  const [regionBarProps] = getComponentPropsFrom(regionBarPath, components)
  return {
    props: {
      // We add an `ssg` field to the page props,
      // which will be provided to the Nextra `useSSG` hook.
      ssg: { regionBarProps },
    },
  }
}

export const RegionBarPropsSection = ({ component }) => {
  // Get the data from SSG, and render it as a component.
  const { regionBarProps } = useSSG()
  return {
    RegionBar: <PropsSection propsList={regionBarProps} />,
  }[component]
}

<header>

# Region Bar

The RegionBar is a custom Button, usually used on mobile devices only. It should trigger the [RegionModal](/components/organisms/region-modal).

The component is part of [Regionalization](/components/feature-regionalization) feature.

</header>

export const RegionBarUsage = () => {
  const { modal, openModal } = useUI()
  return (
    <>
      <RegionBar
        icon={<Icon name="MapPin" />}
        label="Set your location"
        onButtonClick={() => openModal()}
      />
      {modal && <RegionModal fadeOutOnSubmit={true} />}
    </>
  )
}

export const RegionBarEditUsage = () => {
  const { modal, openModal } = useUI()
  return (
    <>
      <RegionBar
        icon={<Icon name="MapPin" />}
        postalCode={'151515'}
        onButtonClick={() => openModal()}
      />
      {modal && <RegionModal fadeOutOnSubmit={true} inputValue={'151515'} />}
    </>
  )
}
