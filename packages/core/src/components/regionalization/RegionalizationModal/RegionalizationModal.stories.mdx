export const ModalTemplate = () => {
  const { modal, openModal } = useUI()
  return (
    <div>
      <RegionalizationButton onClick={() => openModal()}>

      </RegionalizationButton>

    </div>
  )
}

export const Template = (args) => {
  return (

}
