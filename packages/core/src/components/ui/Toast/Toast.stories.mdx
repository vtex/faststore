import Toast from '.'
import Button from 'src/components/ui/Button'

<Meta title="Atoms/Toast" component={Toast} />

export const Template = () => {
  const { pushToast } = useUI()
  return (
    <Button
      variant="tertiary"
      onClick={() => {
        pushToast({
          title: 'Your cart was updated!',
          message: 'Some additional info here.',
          status: 'INFO',
          icon: 'ArrowsClockwise',
        })
      }}
    >
      Display Toast
    </Button>
  )
}

export const TemplateWithProvider = () => {
  return (
    <UIProvider>
      <Toast />
      <Template />
    </UIProvider>
  )
}
