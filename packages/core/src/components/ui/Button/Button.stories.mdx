## Usage

export const Usage = ({ inverse, ...args }) => {
  return inverse ? (
    <div style={{ marginBottom: 10, padding: 10, backgroundColor: '#171a1c' }}>
      <Button inverse {...args}>
        Button
      </Button>
    </div>
  ) : (
    <Button {...args}>Button</Button>
  )
}
